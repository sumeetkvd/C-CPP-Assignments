(2). Swappin the vales using call by reference:

Code:


#include <iostream>
using namespace std;


void swap(int &num1, int &num2){
    int temp = num1;
    num1 = num2;
    num2 = temp;
}

int main(){
    int a,b;
    cout<<"Enter values of a and b for swapping: "<<endl;
    cin>>a>>b;
    cout<< "Before swapping the value of a and b are: " << endl;
    cout << "a is: " << a << " " << "b is:" << b <<endl;
    swap(a,b) ;
    cout<< " a is: " << a << " " << " b is: "<< b << endl;
    return 0;
}
