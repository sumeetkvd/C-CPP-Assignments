1. Define a function print that prints array of string with a separator in between. Array of strings and separator should come from parameters and should be default to string “\n”.


Ans.

#include <bits/stdc++.h>
using namespace std;

void string_size(string s, string del = " ")
{
	int start, end = -1*del.size();
	do {
		start = end + del.size();
		end = s.find(del, start);
		cout << s.substr(start, end - start) << endl;
	} while (end != -1);
}

int main(int argc, char const* argv[])
{

	string a = "How$%do$%you$%do$%!";
	string_size(a, "$%");
	cout << endl;

	return 0;
}


2. Write a program to swap two integers using call be reference.
Ans. 

#include<stdio.h>  

void change(int num) {    
    printf("Before adding value inside function num=%d \n",num);    
    num=num+100;    
    printf("After adding value inside function num=%d \n", num);    
}    

int main() {    
    int x=100;    
    printf("Before function call x=%d \n", x);    
    change(x);//passing value in function    
    printf("After function call x=%d \n", x);    
return 0;  
}
